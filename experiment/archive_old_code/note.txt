	activity_count_stage_1, activity_count_stage_2, activity_count_stage_3,
	active_week_ratio_stage_1, active_week_ratio_stage_2, active_week_ratio_stage_3,

	early_warning_stage_1, early_warning_stage_2, early_warning_stage_3, early_warning_any,
